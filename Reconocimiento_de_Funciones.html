class Funcion:
    def __init__(self, expresion):
        # Inicializar atributos de la funci√≥n
        self.expresion = expresion.replace(" ", "")
        self.variables = []
        self.operaciones = 0
        self.grado = 0
        self.constantes = 0

    def analizar(self):
        # M√©todo principal que analiza la expresi√≥n
        exp = self.expresion
        for i in range(len(exp)):
            c = exp[i]
            # Identificar variables (letras)
            if c.isalpha() and c not in self.variables:
                self.variables.append(c)
            # Contar operaciones
            if c in "+-*/^":
                self.operaciones += 1
            # Contar constantes num√©ricas
            if c.isdigit():
                self.constantes += 1
        # Calcular el grado de la funci√≥n
        self.grado = self.calcular_grado()

    def calcular_grado(self):
        # Determinar el mayor exponente en la expresi√≥n
        exp = self.expresion
        grado_max = 0
        i = 0
        while i < len(exp):
            if exp[i].isalpha():
                # Verificar si hay un exponente expl√≠cito (^)
                if i + 1 < len(exp) and exp[i + 1] == '^':
                    j = i + 2
                    num = ''
                    # Extraer el valor del exponente
                    while j < len(exp) and exp[j].isdigit():
                        num += exp[j]
                        j += 1
                    if num != '':
                        grado = int(num)
                        if grado > grado_max:
                            grado_max = grado
                else:
                    # Si no hay exponente, el grado es 1
                    if grado_max < 1:
                        grado_max = 1
            i += 1
        return grado_max

    def resumen(self):
        # Generar un resumen formateado del an√°lisis
        texto = "==============================\n"
        texto += f"Funci√≥n ingresada: {self.expresion}\n"
        texto += f"Variables encontradas: {', '.join(self.variables) if self.variables else 'ninguna'}\n"
        texto += f"N√∫mero de variables: {len(self.variables)}\n"
        texto += f"N√∫mero de operaciones: {self.operaciones}\n"
        texto += f"N√∫mero de constantes: {self.constantes}\n"
        texto += f"Grado de la funci√≥n: {self.grado}\n"
        
        # Clasificar el tipo de funci√≥n seg√∫n su grado
        if self.grado == 1:
            tipo = "Funci√≥n lineal"
        elif self.grado == 2:
            tipo = "Funci√≥n cuadr√°tica"
        elif self.grado == 3:
            tipo = "Funci√≥n c√∫bica"
        elif self.grado > 3:
            tipo = "Funci√≥n polin√≥mica de grado mayor"
        else:

        texto += f"Tipo de funci√≥n: {tipo}\n"
