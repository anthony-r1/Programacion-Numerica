def graficar_lineas_ascii():
    try:
        restric1 = input("Ingrese la primera restricci√≥n (ej: x+y<=15): ").replace(" ", "")
        restric2 = input("Ingrese la segunda restricci√≥n (ej: 3x+5y<=20): ").replace(" ", "")
        
        def parse_restric(eq):
            # Detectar vertical (x=...)
            if eq.startswith("x<=") or eq.startswith("x>=") or eq.startswith("x="):
                num = float(eq.split("=")[1])
                return "vertical", num
            
            # Detectar horizontal (y=...)
            if eq.startswith("y<=") or eq.startswith("y>=") or eq.startswith("y="):
                num = float(eq.split("=")[1])
                return "horizontal", num
            
            # Caso general ax + by + c = 0
            if 'x' not in eq and 'y' not in eq:
                raise ValueError("Ecuaci√≥n inv√°lida.")
            
            # Separar en lados izquierdo y derecho
            if "<=" in eq:
                lhs, rhs = eq.split("<=")
            elif ">=" in eq:
                lhs, rhs = eq.split(">=")
            elif "=" in eq:
                lhs, rhs = eq.split("=")
            else:
                raise ValueError("Restricci√≥n no v√°lida.")
            
            rhs = float(rhs)
            a, b = 0, 0
            for part in lhs.replace("-", "+-").split("+"):
                if part == "":
                    continue
                if "x" in part:
                    coef = part.replace("x", "")
                    a = 1 if coef in ("", "+") else -1 if coef == "-" else float(coef)
                elif "y" in part:
                    coef = part.replace("y", "")
                    b = 1 if coef in ("", "+") else -1 if coef == "-" else float(coef)
            
            return "normal", (a, b, rhs)
        
        tipo1, datos1 = parse_restric(restric1)

        print("\nLeyenda:\n * = R1\n o = R2\n # = Intersecci√≥n\n | = Eje Y\n - = Eje X\n + = Origen")
        
        for y in range(15, -16, -1):
            linea = ""
            for x in range(-15, 16):
                def cumple(tipo, datos):
                    if tipo == "vertical":
                        return x == datos
                    elif tipo == "horizontal":
                        return y == datos
                    else:
                        a, b, c = datos
                        if b != 0:
                            y_calc = (c - a * x) / b
                            return round(y_calc) == y

                            x_calc = c / a
                            return round(x_calc) == x

                on_r1 = cumple(tipo1, datos1)

                if on_r1 and on_r2:

                    linea += "*"

                    linea += "o"
                elif x == 0 and y == 0:

                elif x == 0:
